ssh bob@node01
cd /etc/netplan
ls -la

vi /etc/netplan/99-bridge.yaml
network:
  version: 2
  renderer: networkd
  ethernets:
    eth1:
      dhcp4: no
    eth2:
      dhcp4: no

    bridge1:
      dhcp4: yes
